45 /* static */ std::string ResponseSink::EscapeArgument(const std::string &argument)
46 {
47  bool escaping = false;
48  std::string escaped;
49 
50  for (unsigned char c : argument) {
51  // These are the characters (including all whitespace, via
52  // isspace()) whose presence means we need to single-quote
53  // escape the argument.
54  bool is_escaper = c == '"' || c == '\'' || c == '\\';
55  if (isspace(c) || is_escaper) escaping = true;
56 
57  // Since we use single-quote escaping, the only thing we need
58  // to escape by itself is single quotes, which are replaced by
59  // the sequence '\'' (break out of single quotes, escape a
60  // single quote, then re-enter single quotes).
61  escaped += (c == '\'') ? R"('\'')" : std::string(1, c);
62  }
63 
64  // Only single-quote escape if necessary.
65  // Otherwise, it wastes two characters!
66  if (escaping) return "'" + escaped + "'";
67  return escaped;
68 }
