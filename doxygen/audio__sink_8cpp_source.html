77 void AudioSink::Transfer(AudioSink::TransferIterator &start,
78  const AudioSink::TransferIterator &end)
79 {
80  // No point transferring 0 bytes.
81  if (start == end) {
82  return;
83  }
84  assert(start < end);
85 
86  unsigned long bytes = std::distance(start, end);
87  // There should be a whole number of samples being transferred.
88  assert(bytes % bytes_per_sample == 0);
89  assert(0 < bytes);
90 
91  auto samples = bytes / this->bytes_per_sample;
92 
93  // Only transfer as many samples as the ring buffer can take.
94  // Don't bother trying to write 0 samples!
95  auto count = std::min(samples, this->ring_buf.WriteCapacity());
96  if (count == 0) {
97  return;
98  }
99  assert(0 < count);
100 
101  auto start_ptr = reinterpret_cast<char *>(&*start);
102  unsigned long written_count = this->ring_buf.Write(start_ptr, count);
103  // Since we never write more than the ring buffer can take, the written
104  // count should equal the requested written count.
105  assert(written_count == count);
106 
107  start += (written_count * this->bytes_per_sample);
108  assert(start <= end);
109 }

112  unsigned long frames_per_buf,

114  PaStreamCallbackFlags)

127 PlayCallbackStepResult AudioSink::PlayCallbackStep(char *out,
128  unsigned long frames_per_buf,
129  PlayCallbackStepResult in)
130 {
131  unsigned long avail = this->ring_buf.ReadCapacity();
132  bool empty = avail == 0;
133 
134  /* If we've just started this stream, we don't want to hand PortAudio
135  an incomplete frame—we'd rather wait until we have enough in the
136  ring buffer before starting to play out. */
137  bool wait = this->just_started && (avail < frames_per_buf);
138 
139  bool failed = wait || empty;
140  auto fn = failed ? &AudioSink::PlayCallbackFailure
141  : &AudioSink::PlayCallbackSuccess;
142  return (this->*fn)(out, avail, frames_per_buf, in);
143 }

157 PlayCallbackStepResult AudioSink::PlayCallbackFailure(
158  char *out, unsigned long, unsigned long frames_per_buf,
159  PlayCallbackStepResult in)
160 {
161  decltype(in) result;
162 
163  if (InputReady()) {
164  // There's been some sort of genuine issue.
165  // Make up some silence to plug the gap.
166  Debug() << "Buffer underflow" << std::endl;
167  memset(out, 0, this->bytes_per_sample * frames_per_buf);
168  result = std::make_pair(paContinue, frames_per_buf);
169  } else {
170  // End of input is ok, it means the stream can finish.
171  result = std::make_pair(paComplete, in.second);
172  }
173 
174  return result;
175 }

177 unsigned long AudioSink::ReadSamplesToOutput(char *&output,
178  unsigned long output_capacity,
179  unsigned long buffered_count)
180 {
181  // Transfer the maximum that we can offer to PortAudio without
182  // overshooting
183  // its sample request limit.
184  long transfer_sample_count = static_cast<long>(
185  std::min({ output_capacity, buffered_count,
186  static_cast<unsigned long>(LONG_MAX) }));
187  output += this->ring_buf.Read(output, transfer_sample_count);
188 
189  // Update the position count so it reflects the last position that was
190  // sent
191  // for playback (*not* the last position decoded).
192  this->position_sample_count += transfer_sample_count;
193  return transfer_sample_count;
194 }
