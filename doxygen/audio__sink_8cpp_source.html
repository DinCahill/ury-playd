76 void AudioSink::Transfer(AudioSink::TransferIterator &start,
77  const AudioSink::TransferIterator &end)
78 {
79  // No point transferring 0 bytes.
80  if (start == end) {
81  return;
82  }
83  assert(start < end);
84 
85  unsigned long bytes = std::distance(start, end);
86  // There should be a whole number of samples being transferred.
87  assert(bytes % bytes_per_sample == 0);
88  assert(0 < bytes);
89 
90  auto samples = bytes / this->bytes_per_sample;
91 
92  // Only transfer as many samples as the ring buffer can take.
93  // Don't bother trying to write 0 samples!
94  auto count = std::min(samples, this->ring_buf.WriteCapacity());
95  if (count == 0) {
96  return;
97  }
98  assert(0 < count);
99 
100  auto start_ptr = reinterpret_cast<char *>(&*start);
101  unsigned long written_count = this->ring_buf.Write(start_ptr, count);
102  // Since we never write more than the ring buffer can take, the written
103  // count should equal the requested written count.
104  assert(written_count == count);
105 
106  start += (written_count * this->bytes_per_sample);
107  assert(start <= end);
108 }

111  unsigned long frames_per_buf,
112  const PaStreamCallbackTimeInfo *,
113  PaStreamCallbackFlags)
114 {
115  char *cout = static_cast<char *>(out);
116 
117  std::pair<PaStreamCallbackResult, unsigned long> result =

126 PlayCallbackStepResult AudioSink::PlayCallbackStep(char *out,
127  unsigned long frames_per_buf,
128  PlayCallbackStepResult in)
129 {
130  unsigned long avail = this->ring_buf.ReadCapacity();
131  bool empty = avail == 0;
132 
133  /* If we've just started this stream, we don't want to hand PortAudio
134  an incomplete frame—we'd rather wait until we have enough in the
135  ring buffer before starting to play out. */
136  bool wait = this->just_started && (avail < frames_per_buf);
137 
138  bool failed = wait || empty;
139  auto fn = failed ? &AudioSink::PlayCallbackFailure
140  : &AudioSink::PlayCallbackSuccess;
141  return (this->*fn)(out, avail, frames_per_buf, in);
142 }

156 PlayCallbackStepResult AudioSink::PlayCallbackFailure(
157  char *out, unsigned long, unsigned long frames_per_buf,
158  PlayCallbackStepResult in)
159 {
160  decltype(in) result;
161 
162  if (InputReady()) {
163  // There's been some sort of genuine issue.
164  // Make up some silence to plug the gap.
165  Debug() << "Buffer underflow" << std::endl;
166  memset(out, 0, this->bytes_per_sample * frames_per_buf);
167  result = std::make_pair(paContinue, frames_per_buf);
168  } else {
169  // End of input is ok, it means the stream can finish.
170  result = std::make_pair(paComplete, in.second);
171  }
172 
173  return result;
174 }

176 unsigned long AudioSink::ReadSamplesToOutput(char *&output,
177  unsigned long output_capacity,
178  unsigned long buffered_count)
179 {
180  // Transfer the maximum that we can offer to PortAudio without
181  // overshooting
182  // its sample request limit.
183  long transfer_sample_count = static_cast<long>(
184  std::min({ output_capacity, buffered_count,
185  static_cast<unsigned long>(LONG_MAX) }));
186  output += this->ring_buf.Read(output, transfer_sample_count);
187 
188  // Update the position count so it reflects the last position that was
189  // sent
190  // for playback (*not* the last position decoded).
191  this->position_sample_count += transfer_sample_count;
192  return transfer_sample_count;
193 }
