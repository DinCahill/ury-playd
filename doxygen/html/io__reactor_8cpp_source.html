114 void IoReactor::NewConnection(uv_stream_t *server)
115 {
116  uv_tcp_t *client = new uv_tcp_t();
117  uv_tcp_init(uv_default_loop(), client);
118 
119  if (uv_accept(server, (uv_stream_t *)client) == 0) {
120  Debug() << "New connection" << std::endl;
121  auto tcp = std::make_shared<TcpResponseSink>(*this, client,
122  this->handler);
123  this->player.WelcomeClient(*tcp);
124  this->connections.insert(tcp);
125  client->data = static_cast<void *>(tcp.get());
126 
127  uv_read_start((uv_stream_t *)client, UvAlloc, UvReadCallback);
128  } else {
129  uv_close((uv_handle_t *)client, UvCloseCallback);
130  }
131 }

198 void TcpResponseSink::RespondRaw(const std::string &string) const
199 {
200  Debug() << "Sending command:" << string << std::endl;
201  unsigned int l = string.length();
202  const char *s = string.c_str();
203 
204  WriteReq *req = new WriteReq;

206  memcpy(req->buf.base, s, l);
