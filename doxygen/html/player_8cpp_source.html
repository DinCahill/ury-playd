89 bool Player::Load(const std::string &path)
90 {
91  bool valid = !path.empty();
92  if (valid) {

94  {
95  this->file.Load(path);
96  ResetPosition();

105  catch (Error &)

134 bool Player::Seek(const std::string &time_str)
135 {
136  if (CurrentStateIn(PlayerState::AUDIO_LOADED_STATES)) {
137  bool success = true;
138  std::chrono::microseconds position(0);
139 
140  try

144  catch (std::out_of_range)
145  {
146  success = false;
147  }
148 
149  if (success) {
150  this->file.SeekToPosition(position);
151  this->ResetPosition();
152  this->UpdatePosition();
153  }
154  return success;
