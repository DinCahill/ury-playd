124 void ConnectionPool::Accept(uv_stream_t *server)
125 {
126  uv_tcp_t *client = new uv_tcp_t();
127  uv_tcp_init(uv_default_loop(), client);
128 
129  if (uv_accept(server, (uv_stream_t *)client) == 0) {
130  this->connections.emplace_back(
131  new Connection(*this, client, this->handler));

134  client->data = static_cast<void *>(
135  this->connections.back().get());
136 
137  uv_read_start((uv_stream_t *)client, UvAlloc, UvReadCallback);
138  } else {

140  }

190 void IoCore::InitAcceptor(const std::string &address, const std::string &port)
191 {
192  int uport = std::stoi(port);
193 
194  uv_tcp_init(uv_default_loop(), &this->server);
195  this->server.data = static_cast<void *>(&this->pool);
196 
197  struct sockaddr_in bind_addr;
198  uv_ip4_addr(address.c_str(), uport, &bind_addr);
199  uv_tcp_bind(&this->server, (const sockaddr *)&bind_addr, 0);
200 
201  int r = uv_listen((uv_stream_t *)&this->server, 128, UvListenCallback);
202  if (r) {

204  " (" + std::string(uv_err_name(r)) + ")");
205  }
206 
207  Debug() << "Listening at" << address << "on" << port << std::endl;
208 }

237 void Connection::RespondRaw(const std::string &string) const
238 {
239  unsigned int l = string.length();
240  const char *s = string.c_str();
241 
242  WriteReq *req = new WriteReq;

247  uv_write((uv_write_t *)req, (uv_stream_t *)this->tcp, &req->buf, 1,

251 std::string Connection::Name()
252 {
253  // Warning: fairly low-level Berkeley sockets code ahead!
254  // (Thankfully, libuv makes sure the appropriate headers are included.)
255 
256  // Using this instead of struct sockaddr is advised by the libuv docs,
257  // for IPv6 compatibility.
258  struct sockaddr_storage s;
259  auto sp = (struct sockaddr *)&s;
260 
261  // Turns out if you don't do this, Windows (and only Windows?) is upset.
262  socklen_t namelen = sizeof(s);
263 
264  std::ostringstream os;
265 
266  int pe = uv_tcp_getpeername(this->tcp, sp, (int *)&namelen);
267  if (pe) {
268  os << "<error (peer): " << uv_strerror(pe) << ">";
269  return os.str();
270  }
271 
272  // Now, split the sockaddr into host and service.
273  char host[NI_MAXHOST];
274  char serv[NI_MAXSERV];
275 
276  // We use NI_NUMERICSERV to ensure a port number comes out.
277  // Otherwise, we could get a (likely erroneous) string description of
278  // what the network stack *thinks* the port is used for.
279  int ne = getnameinfo(sp, namelen, host, sizeof(host), serv,
280  sizeof(serv), NI_NUMERICSERV);
281  if (ne) {
282  os << "<error (name): " << gai_strerror(ne) << ">";
283  return os.str();
284  }
285 
286  os << host << ":" << serv;
287  return os.str();
288 }

290 void Connection::Read(ssize_t nread, const uv_buf_t *buf)
291 {
292  // Did the connection hang up? If so, de-pool it.
293  // De-pooling the connection will usually lead to the connection being
294  // destroyed.
295  if (nread == UV_EOF) {
296  this->Depool();
297  return;
298  }
299 
300  // Did we hit any other read errors? Also de-pool, but log the error.
301  if (nread < 0) {
302  Debug() << "Error on" << Name() << "-" << uv_err_name(nread)
303  << std::endl;
304  this->Depool();
305  return;
306  }
307 
308  auto chars = buf->base;
309 
310  // Make sure we actually have some data to read!
311  if (chars == nullptr) return;
312 
313  // Everything looks okay for reading.
314  auto lines = this->tokeniser.Feed(std::string(chars, nread));
315  for (auto line : lines) HandleCommand(line);
316  delete[] chars;
317 }
