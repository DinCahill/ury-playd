86 size_t AudioSource::BytesPerSample() const
87 {
88  assert(this->context != nullptr);
89 
90  // Since libsox always outputs 32-bit samples, the bytes per sample is
91  // always 4 per channel.
92 
93  // SoX has a slightly peculiar notion of sample counts, in that it
94  // regards each channel as having its own separate sample, so we need
95  // to multiply and divide sample counts by the channel count when
96  // talking to SoX.
97  return 4 * this->ChannelCount();
98 }

100 std::uint64_t AudioSource::Seek(TimeParser::MicrosecondPosition position)
101 {
102  assert(this->context != nullptr);
103 
104  auto samples = this->SamplePositionFromMicroseconds(position);
105 
106  // See BytesPerSample() for an explanation of this ChannelCount().
107  auto sox_samples = samples * this->ChannelCount();
108 
109  // libsox doesn't seem to like seeking into an ended file, so close
110  // and re-open it first.
111  if (this->decode_state == DecodeState::END_OF_FILE) {
112  std::string path = this->Path();
113  this->Close();
114  this->Open(path);
115  }
116 
117  // Have we tried to seek past the end of the file?
118  if (this->context->signal.length < sox_samples) {
119  throw SeekError(MSG_SEEK_FAIL);
120  }
121 
122  if (sox_seek(this->context, sox_samples, SOX_SEEK_SET) != SOX_SUCCESS) {
123  throw SeekError(MSG_SEEK_FAIL);
124  }
125 
126  // Reset the decoder state, because otherwise the decoder will get very
127  // confused.
128  this->decode_state = DecodeState::DECODING;
129 
130  return samples;
131 }

133 AudioSource::DecodeResult AudioSource::Decode()
134 {
135  assert(this->context != nullptr);
136 
137  auto buf = reinterpret_cast<sox_sample_t *>(&this->buffer.front());
138 
139  // See BytesPerSample() for an explanation of this ChannelCount().
140  auto sox_capacity = this->BufferSampleCapacity() * this->ChannelCount();
141  size_t read = sox_read(this->context, buf, sox_capacity);
142 
143  DecodeVector decoded;
144 
145  if (read == 0) {
146  this->decode_state = DecodeState::END_OF_FILE;
147  } else {
148  this->decode_state = DecodeState::DECODING;
149 
150  // Copy only the bit of the buffer occupied by decoded data
151  // See BytesPerSample() for an explanation of the
152  // ChannelCount() division.
153  auto front = this->buffer.begin();
154  auto read_bytes = (this->BytesPerSample() * read) /
155  this->ChannelCount();
156  decoded = DecodeVector(front, front + read_bytes);
157  }
158 
159  return std::make_pair(this->decode_state, decoded);
160 }
