82 void Audio::TransferFrame()
83 {
84  assert(!this->frame.empty());
85 
86  this->sink->Transfer(this->frame_iterator, this->frame.end());
87 
88  // We empty the frame once we're done with it. This
89  // maintains FrameFinished(), as an empty frame is a finished one.
90  if (FrameFinished()) {

92  assert(FrameFinished());

94 
95  // The frame iterator should be somewhere between the beginning and
96  // end of the frame, unless the frame was emptied.
97  assert(this->frame.empty() ||

102 bool Audio::DecodeIfFrameEmpty()

106  // frame as soon as it finishes.

108

110

115  this->frame_iterator = this->frame.begin();

119  }
