92 void PipeAudio::TransferFrame()
93 {
94  assert(!this->frame.empty());
95  assert(this->sink != nullptr);
96  assert(this->src != nullptr);
97 
98  this->sink->Transfer(this->frame_iterator, this->frame.end());
99 
100  // We empty the frame once we're done with it. This
101  // maintains FrameFinished(), as an empty frame is a finished one.
102  if (this->FrameFinished()) {
103  this->ClearFrame();
104  assert(this->FrameFinished());
105  }
106 
107  // The frame iterator should be somewhere between the beginning and
108  // end of the frame, unless the frame was emptied.
109  assert(this->frame.empty() ||
110  (this->frame.begin() <= this->frame_iterator &&
111  this->frame_iterator < this->frame.end()));
112 }

114 bool PipeAudio::DecodeIfFrameEmpty()
115 {
116  // Either the current frame is in progress, or has been emptied.
117  // AdvanceFrameIterator() establishes this assertion by emptying a
118  // frame as soon as it finishes.
119  assert(this->frame.empty() || !this->FrameFinished());
120 
121  // If we still have a frame, don't bother decoding yet.
122  if (!this->FrameFinished()) return true;
123 
124  assert(this->src != nullptr);
125  AudioSource::DecodeResult result = this->src->Decode();
126 
127  this->frame = result.second;
128  this->frame_iterator = this->frame.begin();
129 
130  return result.first != AudioSource::DecodeState::END_OF_FILE;
131 }
