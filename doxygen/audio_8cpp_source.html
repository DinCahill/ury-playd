83 void Audio::TransferFrame()
84 {
85  assert(!this->frame.empty());
86 
87  this->sink->Transfer(this->frame_iterator, this->frame.end());
88 
89  // We empty the frame once we're done with it. This
90  // maintains FrameFinished(), as an empty frame is a finished one.
91  if (this->FrameFinished()) {
92  this->ClearFrame();
93  assert(this->FrameFinished());
94  }
95 
96  // The frame iterator should be somewhere between the beginning and
97  // end of the frame, unless the frame was emptied.
98  assert(this->frame.empty() ||
99  (this->frame.begin() <= this->frame_iterator &&
100  this->frame_iterator < this->frame.end()));
101 }
