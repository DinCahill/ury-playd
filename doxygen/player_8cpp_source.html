77 void Player::End()
78 {
79  if (this->end_sink != nullptr) {
80  this->end_sink->RespondArgs(ResponseCode::END, {});
81  }
82  this->Stop();
83 
84  // Rewind the file back to the start. We can't use Seek() here
85  // in case End() is called from Seek(); a seek failure could start an
86  // infinite loop.
87  this->file.SeekToPosition(0);
88  this->ResetPosition();
89  this->UpdatePosition();
90 }

108 CommandResult Player::Load(const std::string &path)
109 {
110  if (path.empty()) return CommandResult::Invalid(MSG_LOAD_EMPTY_PATH);
111 
112  try {
113  this->file.Load(path);
114  this->ResetPosition();
115  this->SetState(PlayerState::State::STOPPED);
116  }
117  catch (FileError &e) {
118  // File errors aren't fatal, so catch them here.
119  this->Eject();
120  return CommandResult::Failure(e.Message());
121  }
122  catch (Error &) {
123  // Ensure a load failure doesn't leave a corrupted track
124  // loaded.
125  this->Eject();
126  throw;
127  }
128 
129  return CommandResult::Success();
130 }

153 CommandResult Player::Seek(const std::string &time_str)
154 {
155  if (!this->CurrentStateIn(PlayerState::AUDIO_LOADED_STATES)) {
156  return CommandResult::Invalid(MSG_CMD_NEEDS_LOADED);
157  }
158 
159  TimeParser::MicrosecondPosition position(0);
160 
161  try {
162  position = this->time_parser.Parse(time_str);
163  }
164  catch (std::out_of_range) {
165  return CommandResult::Invalid(MSG_SEEK_INVALID_UNIT);
166  }
167  catch (SeekError) {
168  return CommandResult::Invalid(MSG_SEEK_INVALID_VALUE);
169  }
170 
171  try {
172  this->file.SeekToPosition(position);
173  }
174  catch (SeekError) {
175  Debug() << "Seek failure" << std::endl;
176 
177  // Make it look to the client as if the seek ran off the end of
178  // the file.
179  this->End();
180  return CommandResult::Success();
181  }
182 
183  // Clean out the position tracker, as the position has abruptly changed.
184  this->ResetPosition();
185  this->UpdatePosition();
186 
187  return CommandResult::Success();
188 }
