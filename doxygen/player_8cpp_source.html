37 bool Player::Update()
38 {
39  auto as = this->file.Update();
40  if (as == Audio::State::AT_END) this->End();
41  if (as == Audio::State::PLAYING) {
42  // Since the audio is currently playing, the position may have
43  // advanced since last update. So we need to update it.
44  this->position.Update(this->file.Position());
45  }
46 
47  return this->state.IsRunning();
48 }

60 {

88 CommandResult Player::Load(const std::string &path)
89 {
90  if (path.empty()) return CommandResult::Invalid(MSG_LOAD_EMPTY_PATH);
91 
92  try {
93  this->file.Load(path);
94  this->position.Reset();
95  this->state.Set(PlayerState::State::STOPPED);
96  } catch (FileError &e) {
97  // File errors aren't fatal, so catch them here.
98  this->Eject();
99  return CommandResult::Failure(e.Message());
100  } catch (Error &) {
101  // Ensure a load failure doesn't leave a corrupted track
102  // loaded.
103  this->Eject();
104  throw;
105  }
106 
107  return CommandResult::Success();
108 }

131 CommandResult Player::Seek(const std::string &time_str)
132 {
133  if (!this->state.In(PlayerState::AUDIO_LOADED_STATES)) {
134  return CommandResult::Invalid(MSG_CMD_NEEDS_LOADED);
135  }
136 
137  std::uint64_t pos = 0;
138  size_t cpos = 0;
139  try {
140  // In previous versions, this used to parse a unit at the end.
141  // This was removed for simplicity--use baps3-cli etc. instead.
142  pos = std::stoull(time_str, &cpos);
143  } catch (...) {
144  // Should catch std::out_of_range and std::invalid_argument.
145  // http://www.cplusplus.com/reference/string/stoull/#exceptions
146  return CommandResult::Invalid(MSG_SEEK_INVALID_VALUE);
147  }
148 
149  // cpos will point to the first character in pos that wasn't a number.
150  // We don't want any characters here, so bail if the position isn't at
151  // the end of the string.
152  auto sl = time_str.length();
153  if (cpos != sl) return CommandResult::Invalid(MSG_SEEK_INVALID_VALUE);
154 
155  try {
156  this->SeekRaw(pos);
157  } catch (SeekError) {
158  Debug() << "Seek failure" << std::endl;
159 
160  // Make it look to the client as if the seek ran off the end of
161  // the file.
162  this->End();
163  }
164 
165  return CommandResult::Success();
166 }
