39 int Playd::GetDeviceID()
40 {
41  // Did the user provide an ID at all?
42  if (this->arguments.size() < 2) return this->INVALID_ID;
43 
44  // Only accept valid numbers (stoi will throw for invalid ones).
45  int id;
46  try {
47  id = std::stoi(this->arguments[1]);
48  }
49  catch (...) {
50  // Only std::{invalid_argument,out_of_range} are thrown here.
51  return this->INVALID_ID;
52  }
53 
54  // Only allow valid, outputtable devices; reject input-only devices.
55  if (!this->audio.IsOutputDevice(id)) return this->INVALID_ID;
56 
57  return id;
58 }
