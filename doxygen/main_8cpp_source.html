130 int playd::Run()
131 {
132  try

134  // Don't roll this into the constructor: it'll go out of scope!
135  int id = this->GetDeviceID();
136  if (id == -1) {
137  /* Oops, user entered an invalid sound device. */
138  auto device_list = this->audio.GetDevicesInfo();
139  for (const auto &device : device_list) {
140  std::cout << device.first << ": "
141  << device.second << std::endl;
142  }
143  return EXIT_FAILURE;
144  }
145  this->audio.SetDeviceID(id);
146 
147  this->player.SetPositionResponsePeriod(POSITION_PERIOD);
148  this->player.SetResponseSink(*this->io);
